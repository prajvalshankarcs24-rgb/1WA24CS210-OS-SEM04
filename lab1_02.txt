#include <stdio.h>
int main() {
    int n;
    printf("Enter number of processes: ");
    scanf("%d", &n);
    int pid[n], at[n], bt[n], ct[n], tat[n], wt[n], rt[n], finished[n];
    for (int i = 0; i < n; i++) {
        printf("Enter ProcessID, ArrivalTime, BurstTime: ");
        scanf("%d %d %d", &pid[i], &at[i], &bt[i]);
        rt[i] = bt[i];
        finished[i] = 0;
    }
    int completed = 0, current_time = 0;
    while (completed < n) {
        int idx = -1;
        int min_remaining = 99999;
        for (int i = 0; i < n; i++) {
            if (!finished[i] && at[i] <= current_time && rt[i] < min_remaining) {
                min_remaining = rt[i];
                idx = i;
            }
        }
        if (idx == -1) {
            current_time++;
        } else {
            rt[idx]--;
            current_time++;
            if (rt[idx] == 0) {
                ct[idx] = current_time;
                tat[idx] = ct[idx] - at[idx];
                wt[idx] = tat[idx] - bt[idx];
                finished[idx] = 1;
                completed++;
            }
        }
    }
    printf("\nPID\tAT\tBT\tCT\tTAT\tWT\n");
    for (int i = 0; i < n; i++) {
        printf("%d\t%d\t%d\t%d\t%d\t%d\n",
               pid[i], at[i], bt[i], ct[i], tat[i], wt[i]);
    }
    return 0;
}
