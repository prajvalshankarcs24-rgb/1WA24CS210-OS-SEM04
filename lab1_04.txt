// Write a C program to find if there are duplicates in an array.
#include<stdio.h>
#include<stdlib.h>
int main(){
    int target,size,count=0;
    printf("Enter the size of the array: ");
    scanf("%d",&size);
    int arr[size];
    printf("Enter the elements of the array: \n");
    for(int i=0;i<size;i++){
        scanf("%d",&arr[i]);
    }
    for (int i=0; i<size; i++) {
            for (int j=i; j<size; j++) {
                if (arr[j] == arr[i]) {
                    count = count+1;
                }
            }
            if (count>1) {
                printf("Duplicate element: %d\n", arr[i]);
            }
            count=0;
    }
    return 0;
}
