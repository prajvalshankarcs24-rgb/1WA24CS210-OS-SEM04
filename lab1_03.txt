// Write a C program to perform Binary search
#include <stdio.h>
int main() {
    int size, key;
    printf("Enter size of array: ");
    scanf("%d", &size);
    int arr[size];
    printf("Enter elements (sorted):\n");
    for (int i = 0; i < size; i++) {
        scanf("%d", &arr[i]);
    }
    printf("Enter element to search: ");
    scanf("%d", &key);
    int low = 0, high = size - 1, mid, found = -1;
    while (low <= high) {
        mid = (low + high) / 2;
        if (arr[mid] == key) {
            found = mid;
            break;
        } else if (arr[mid] < key) {
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    if (found != -1)
        printf("Element found at position %d\n", found + 1);
    else
        printf("Element not found\n");
    return 0;
}
